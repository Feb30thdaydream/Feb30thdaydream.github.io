由先序和中序求后序二叉树

2022-06-20 1 min read # study 
BinTree Build(int pre[] ,int in[] ,int size)
{
	if(size<=0)
        return NULL;
	//先在中序中找到根节点
	int i;
	for(i=0;i<size;i++)
	{
		if(in[i]==pre[0])break;
	} 
	BinTree tree=(BinTree)malloc(sizeof(struct node));
	tree->date=pre[0];
	tree->left=Build(pre+1,in,i);
	tree->right=Build(pre+i+1,in+i+1,size-1-i);
	return tree;
}
