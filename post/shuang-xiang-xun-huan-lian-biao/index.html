    Node *s=(Node*)malloc(sizeof(Node));
    temp->data=data;
    temp->prior=NULL;
    temp->next=NULL; 
    
    Node * p=head;
    for (int i=1; i<add-1; i++){ /*找到要插入位置的前一个结点*/
            p=p->next;
    }
    s->prior=p->prior;
    p->prior->next=s;
    s->next=p;
    p->prior=s;
}
