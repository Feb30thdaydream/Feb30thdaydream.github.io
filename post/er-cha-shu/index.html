🔸1、先序遍历

void PreOrder(BitTree T){
    if(T==NULL)
        return;
    printf("%d ",T->data);
    PreOrder(T->lchild);//先序递归遍历左子树 
    PreOrder(T->rchild);//先序递归遍历右子树 
}

🔸2、中序遍历

void InOrder(BitTree T){
    if(T==NULL)
        return ;
    InOrder(T->lchild);//中序递归遍历左子树
    printf("%d ",T->data);
    InOrder(T->rchild);//中序递归遍历右子树 
}

🔸3、后序遍历

void PostOrder(BitTree T){
    if(T==NULL)
        return ;
    PostOrder(T->lchild);//后序递归遍历左子树
    PostOrder(T->rchild);//后序递归遍历右子树 
    printf("%d ",T->data);
}

eg.此处以中序遍历为例展示递归过程

🔸3、后序遍历输出叶子节点数目（分治法）

int  PostGetLeave(BitTree T){
    if(T==NULL)
        return  0;
    else if(T->lchild==NULL&&T->rchild==NULL)
        return 1;
    else
        return  (PostGetLeave(T->lchild)+PostGetLeave(T->rchild));
}

🔸5、求二叉树的高度

    a、分沿法求二叉树高度

int MaxOfTwo(int a,int b){//两数较大值 
    if(a>=b)
        return a;
    else
        return b;
}
int GetHeight(BitTree T){//分沿法求二叉树的高度
    if(T)
        return MaxOfTwo(GetHeight(T->lchild),GetHeight(T->rchild))+1);
    else
        return 0;
} 

    b、后序遍历法求二叉树高度

int depth=1；
void  PostGetHeight(BitTree T,int h){
    if(T!=NULL){
        if(h>depth)
            depth=h;
        PostGetHeight(T->lchild,h+1);
        PostGetHeight(T->rchild,h+1);
    }
    return ;
}
